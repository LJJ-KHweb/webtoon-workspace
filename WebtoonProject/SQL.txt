CREATE TABLE "TB_USER" (
	USER_NO NUMBER PRIMARY KEY,
    USER_ID VARCHAR2(30) UNIQUE NOT NULL,
    USER_PWD VARCHAR2(100) NOT NULL,
    USER_NAME VARCHAR2(30) NOT NULL,
    EMAIL VARCHAR2(100) UNIQUE NOT NULL,
    NICKNAME VARCHAR2(50) NOT NULL,
    ENROLL_DATE DATE DEFAULT SYSDATE,
	MODIFY_DATE DATE DEFAULT SYSDATE,
	STATUS CHAR(1) DEFAULT 'N' CHECK(STATUS IN('Y','N'))
);
CREATE SEQUENCE SEQ_TB_USER
START WITH 1
NOCACHE;


CREATE TABLE "TB_WEBTOON" (
    "WEBTOON_ID" NUMBER PRIMARY KEY,
    "TITLE" VARCHAR2(200) NOT NULL,
    "AUTHOR" VARCHAR2(100),
    "PLATFORM" VARCHAR2(50),
    "DAY_OF_WEEK" VARCHAR2(10),
    "GENRE" VARCHAR2(50),
    "DESCRIPTION" VARCHAR2(1000),
    "THUMBNAIL" VARCHAR2(300),
    "CREATE_DATE" DATE DEFAULT SYSDATE
);
CREATE SEQUENCE SEQ_TB_WEBTOON
START WITH 1
NOCACHE;


CREATE TABLE "TB_FAVORITE" (
    "FAVORITE_ID" NUMBER PRIMARY KEY,
    "USER_NO" NUMBER NOT NULL,
    "WEBTOON_ID" NUMBER NOT NULL,
    "CREATE_DATE" DATE DEFAULT SYSDATE,
    FOREIGN KEY ("USER_NO") REFERENCES TB_USER("USER_NO"),
    FOREIGN KEY ("WEBTOON_ID") REFERENCES TB_WEBTOON("WEBTOON_ID")
);

CREATE SEQUENCE SEQ_TB_FAVORITE
START WITH 1
NOCACHE;

CREATE TABLE "TB_REVIEW" (
    "REVIEW_ID" NUMBER PRIMARY KEY,
    "USER_NO" NUMBER NOT NULL,
    "WEBTOON_ID" NUMBER NOT NULL,
    "RATING" NUMBER,
    "CONTENT" VARCHAR2(1000),
    "CREATE_DATE" DATE DEFAULT SYSDATE,
    FOREIGN KEY ("USER_NO") REFERENCES TB_USER("USER_NO"),
    FOREIGN KEY ("WEBTOON_ID") REFERENCES TB_WEBTOON("WEBTOON_ID")
);

CREATE SEQUENCE SEQ_TB_REVIEW
START WITH 1
NOCACHE;




CREATE TABLE "TB_VIEW_HISTORY" (
    "VIEW_ID" NUMBER PRIMARY KEY,
    "USER_NO" NUMBER,
    "WEBTOON_ID" NUMBER,
    "VIEW_DATE" DATE DEFAULT SYSDATE,
    FOREIGN KEY ("USER_NO") REFERENCES TB_USER("USER_NO"),
    FOREIGN KEY ("WEBTOON_ID") REFERENCES TB_WEBTOON("WEBTOON_ID")
);

CREATE SEQUENCE SEQ_TB_VIEW_HISTORY
START WITH 1
NOCACHE;
